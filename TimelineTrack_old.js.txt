"use strict";

import { timelineConfig } from "./timelineConfig.js";
import { clipConfig } from "./clipConfig.js";
import { TimelineClip } from "./TimelineClip.js";
import { ContextMenu } from "./ContextMenu.js";

/**
 * X軸方向の衝突判定
 * @param {TimelineClip} tclip1 
 * @param {TimelineClip} tclip2 
 * @returns {boolean}
 */
function isCollisionX(tclip1, tclip2){
    if(tclip1 == undefined || tclip2 == undefined){
        return false;
    }

    if(
        (tclip1.startTime <= tclip2.startTime && tclip1.startTime+tclip1.duration >= tclip2.startTime) ||
        (tclip2.startTime <= tclip1.startTime && tclip2.startTime+tclip2.duration >= tclip1.startTime)
    ){
        return true;
    }
    else{
        return false;
    }
}



export class TimelineTrack extends EventTarget{
    /**
     * @param {Timeline} timeline
     * @param {Object} config
     * @param {string} [config.backgroundColor]
     * @param {TrackHeader} header
     */
    constructor(timeline, config, header){
        super();

        const canvas = document.createElement('canvas');
        canvas.classList.add("track-canvas");
        canvas.width = timelineConfig.trackWidth;
        canvas.height = timelineConfig.trackHeight;

        const ctx = canvas.getContext("2d")
        this.timeline = timeline;
        this.canvas = canvas;
        this.ctx = ctx;
        this.width = canvas.width;
        this.height = canvas.height;
        this.config = config;
        this.header = header;

        /** @property {TimelineClip[]} this.tclips */
        this.tclips = [];

        /** @type {TimelineClip | null} */
        this.targetClip = null;
        /** @type {boolean} */
        let startlineResize = false;//スタートラインにマウスがいるか
        /** @type {boolean} */
        let endlineResize = false;
        /**
         * マウスポインタとclip startTimeの距離[px]
         * @type {number} pointer_starttime_distance
         */
        let pointer_starttime_distance = 0;

        // ---マウスイベントの登録---
        canvas.addEventListener("pointerdown", (e) => {
            this.dispatchEvent(new CustomEvent("pointerDown"));

            // マウスの座標をCanvas内の座標とあわせるため
            const rect = canvas.getBoundingClientRect();
            const point = {
                x: e.clientX - rect.left,
                y: e.clientY - rect.top
            };
            
            const mag = timeline.magnification;
    
            for(let i=0; i<this.tclips.length; i++){
                const tclip = this.tclips[i];
                const x = tclip.startTime * mag;
                const w = tclip.duration * mag;
    
                if(x <= point.x && point.x <= x + clipConfig.startlineWidth){
                    this.targetClip = tclip;
                    this.targetClip.target = true;//今は使われていないが、色わけなどに
                    startlineResize = true;
    
                    return;
                }
                else if(x+w-clipConfig.endlineWidth <= point.x && point.x <= x+w){
                    this.targetClip = tclip;
                    this.targetClip.target = true;//今は使われていないが、色わけなどに
                    endlineResize = true;
    
                    return;
                }
                else if(x <= point.x && point.x <= x + w){
                    this.targetClip = tclip;
                    this.targetClip.target = true;//今は使われていないが、色わけなどに
                    pointer_starttime_distance = point.x - (this.targetClip.startTime*mag);

                    this.dispatchEvent(new CustomEvent("clipSelect", {detail:{targetClip: this.targetClip}}));
                    this.targetClip.dispatchEvent(new CustomEvent("select"));
                    
                    return;
                }
                else{
                    continue;
                }
            }

            this.dispatchEvent(new CustomEvent("clipBlur"));
        });


        window.addEventListener("pointermove",(e)=>{
            // マウスの座標をCanvas内の座標とあわせるため
            const rect = canvas.getBoundingClientRect();
            const point = {
                x: e.clientX - rect.left,
                y: e.clientY - rect.top
            };
    
            if(this.targetClip == null){
                return;
            }
    
            const mag = timeline.magnification;
    
            if(!startlineResize && !endlineResize){
                //マウスに付いて移動
                this.targetClip.startTime = (point.x - pointer_starttime_distance)/ mag;
            }
            else{
                //リサイズ
                if(startlineResize){
                    const duration = (point.x / mag) - this.targetClip.startTime;//move duration?
                    this.targetClip.startTime += duration;
                    this.targetClip.relativeStartTime += duration;
                    this.targetClip.duration -= duration;
    
                    if(this.targetClip.duration < clipConfig.minDuration){
                        this.targetClip.duration = clipConfig.minDuration;
                    }
                }
                else{//endlineResize:trueのとき
                    const duration = (point.x / mag) - (this.targetClip.startTime + this.targetClip.duration);
                    this.targetClip.duration = this.targetClip.duration + duration;
    
                    if(this.targetClip.duration < clipConfig.minDuration){
                        this.targetClip.duration = clipConfig.minDuration;
                    }
                }
            }
            
            this.refresh();
        });


        canvas.addEventListener("pointermove",(e)=>{
            // マウスの座標をCanvas内の座標とあわせるため
            const rect = canvas.getBoundingClientRect();
            const point = {
                x: e.clientX - rect.left,
                y: e.clientY - rect.top
            };
    
            //カーソル変更
            if(this.targetClip == null){
                const mag = timeline.magnification;
    
                for(let i=0; i<this.tclips.length; i++){
                    const tclip = this.tclips[i];
                    const x = tclip.startTime * mag;
                    const w = tclip.duration * mag;
            
                    if(x <= point.x && point.x <= x + clipConfig.startlineWidth){
                        document.documentElement.style.cursor = "ew-resize";
                        break;
                    }
                    else if(x+w-clipConfig.endlineWidth <= point.x && point.x <= x+w){
                        document.documentElement.style.cursor = "ew-resize";
                        break;
                    }
                    else if(x <= point.x && point.x <= x + w){
                        document.documentElement.style.cursor = "grab";
                        break;
                    }
                    else{
                        document.documentElement.style.cursor = "auto";
                        continue;
                    }
                }
    
                return;
            }
    
            this.refresh();
        });
    
    
        canvas.addEventListener("pointerout",()=>{
            if(this.targetClip == null){
                document.documentElement.style.cursor = "auto";
            }
        });


        window.addEventListener("pointerup",(e)=>{
            if(this.targetClip == null){
                return;
            }
    
            this.#adjustClips();
            this.refresh();
    
            this.targetClip.target = false;
            this.targetClip = null;
            startlineResize = false;
            endlineResize = false;
        });



        // 右クリックメニュー ---------------------
        canvas.addEventListener("contextmenu",(e)=>{
            e.preventDefault();

            // マウスの座標をCanvas内の座標とあわせるため
            const rect = canvas.getBoundingClientRect();
            const point = {
                x: e.clientX - rect.left,
                y: e.clientY - rect.top
            };

            const mag = this.timeline.magnification;

            for(const tclip of this.tclips){
                const x = tclip.startTime * mag;
                const w = tclip.duration * mag;

                if(x <= point.x && point.x <= x + w){
                    const items = [
                        {
                            label: "split",
                            action: ()=>{this.splitClip(tclip, timeline.playhead.time)}
                        },
                        {
                            label: "delete",
                            action: ()=>{this.deleteClip(tclip)}
                        }
                    ];

                    new ContextMenu(items, e.clientX, e.clientY);

                    break;
                }
                else{
                    continue;
                }
            }
        });

    }

    

    splitClip(splitTargetClip, time){
        
        //timeがクリップ上になければ何もしない
        if(!(splitTargetClip.startTime < time &&
            time < splitTargetClip.startTime + splitTargetClip.duration
        )){
            return;
        }
        

        const newClip = splitTargetClip.duplicate();

        newClip.startTime = time;
        newClip.duration = splitTargetClip.startTime + splitTargetClip.duration - time;

        splitTargetClip.duration = time - splitTargetClip.startTime;

        this.#addTimelineClip(newClip);
    }


    deleteClip(tclip){
        const index = this.tclips.indexOf(tclip);
        this.tclips.splice(index, 1);
        this.refresh();
    }


    /**
     * トラックにクリップを追加する。位置は、クリップ内情報で決まる
     * @param {Clip} clip
     * @param {Object} config
     * @param {string} config.middleColor クリップの色
     * @returns {TimelineClip}
     */
    addClip(clip, config){
        const timelineClip = new TimelineClip(clip, config);
        
        this.#addTimelineClip(timelineClip);

        return timelineClip;
    }

    /**
     * @param {TimelineClip} tclip 
     */
    #addTimelineClip(tclip){
        this.tclips.push(tclip);
        this.#sortClips();
        this.refresh();
    }


    // 引数のtargetClipは、必ずしもthis.targetClipと一致するとは限らない
    #adjustClips(){
        this.#sortClips();

        let targetClipIndex;
        for(let i=0; i<this.tclips.length; i++){
            if(this.tclips[i] == this.targetClip){
                targetClipIndex = i;
                break;
            }
        }

        let beforeClip = this.tclips[targetClipIndex -1];
        let afterClip = this.tclips[targetClipIndex +1];

        //targetClipの一つ前と衝突していれば
        if(isCollisionX(this.targetClip, beforeClip)){
            //clipの後ろ半分に先頭があれば
            if((beforeClip.startTime + beforeClip.duration/2) <= this.targetClip.startTime){
                //beforeClipの後ろに移動
                this.targetClip.startTime = beforeClip.startTime + beforeClip.duration;
                if(isCollisionX(this.targetClip, afterClip)){
                    this.#slideAfterClips(targetClipIndex);
                }
            }
            
            //前半分
            else{
                //beforeClipの前に移動
                this.targetClip.startTime = beforeClip.startTime - this.targetClip.duration;

                this.#sortClips();

                for(let i=0; i<this.tclips.length; i++){
                    if(this.tclips[i] == this.targetClip){
                        targetClipIndex = i;
                        break;
                    }
                }

                beforeClip = this.tclips[targetClipIndex -1];
                afterClip = this.tclips[targetClipIndex +1];

                if(isCollisionX(this.targetClip, beforeClip)){
                    this.#slideAfterClips(targetClipIndex-1);
                }
            }
        }


        else if(isCollisionX(this.targetClip, afterClip)){
            //afterClipの前に移動
            this.targetClip.startTime = afterClip.startTime - this.targetClip.duration;

            this.#sortClips(this);

            for(let i=0; i<this.tclips.length; i++){
                if(this.tclips[i] == this.targetClip){
                    targetClipIndex = i;
                    break;
                }
            }

            beforeClip = this.tclips[targetClipIndex -1];
            afterClip = this.tclips[targetClipIndex +1];

            if(isCollisionX(this.targetClip, beforeClip)){
                this.#slideAfterClips(targetClipIndex-1);
            }
        }


        //最初のクリップのstartがマイナスにいれば
        if(this.tclips[0].startTime < 0){
            const slide_duration = 0 - this.tclips[0].startTime;
            this.tclips.forEach(tclip =>{
                tclip.startTime += slide_duration;
            });
        }
    }


    // targetIndexより後ろのclipを全てずらす
    #slideAfterClips(targetClipIndex){
        const targetClip = this.tclips[targetClipIndex];
        //動かす大きさ
        const slide_duration = targetClip.startTime + targetClip.duration - this.tclips[targetClipIndex+1].startTime;

        for(let i=targetClipIndex+1; i<this.tclips.length-targetClipIndex+1; i++){
            if(this.tclips[i] == undefined){
                return;
            }
            this.tclips[i].startTime += slide_duration;
        }
    }

    /**
     * clipを小さい順に並べ替える
     */
    #sortClips(){
        this.tclips.sort((a,b)=>{return (a.startTime < b.startTime ? -1 : 1);});
    }


    refresh(){
        this.ctx.clearRect(0,0, this.width, this.height);

        // 背景色設定
        this.ctx.fillStyle = this.config.backgroundColor || "white";
        this.ctx.fillRect(0,0, this.width, this.height);

        for(const tclip of this.tclips){
            this.#drawClip(tclip);
        }

        if(this.targetClip != null){
            //targetClipをもう一度描画して、前面に表示
            this.#drawClip(this.targetClip);
        }
    }


    #drawClip(tclip){
        const x = tclip.startTime * this.timeline.magnification;//*定数
        const y = 0;
        const w = tclip.duration * this.timeline.magnification;
        const h = this.height;
        const text = tclip.text || "";

        const ctx = this.ctx;

        ctx.fillStyle = tclip.middleColor;
        ctx.fillRect(x, y, w, h);

        ctx.font = "16px arial";
        ctx.fillStyle = "black";
        ctx.textBaseline = "top";
        ctx.fillText(text, x+clipConfig.startlineWidth, 3, w-clipConfig.startlineWidth - clipConfig.endlineWidth);

        ctx.fillStyle =clipConfig.startlineColor;
        ctx.fillRect(x,y, clipConfig.startlineWidth, h);

        ctx.fillStyle = clipConfig.endlineColor;
        ctx.fillRect(x + w - clipConfig.endlineWidth, y, clipConfig.endlineWidth, h);

        //canvas自体の長さを変更
    }


}
